(async () => { try { // 1. 获取凭证 console.log('1. 开始获取凭证...'); const pageResponse = await fetch('https://mooc1.chaoxing.com/mooc-ans/coursestar', { headers: { 'Accept': '*/*', 'Accept-Language': 'zh-CN,zh;q=0.9', 'Cache-Control': 'no-cache', 'Pragma': 'no-cache' } }); const html = await pageResponse.text(); // 提取凭证信息 const timeMatch = html.match(/window\["currentTime"\]\s*=\s*'([^']*)'/); const encMatch = html.match(/window\["uploadEnc"\]\s*=\s*'([^']*)'/); const credentials = { time: timeMatch?.[1], enc: encMatch?.[1] }; console.log('2. 凭证:', credentials); if (!timeMatch || !encMatch) { throw new Error('获取凭证失败'); } // 2. 选择文件 const input = document.createElement('input'); input.type = 'file'; input.accept = 'image/*'; const file = await new Promise(resolve => { input.onchange = e => resolve(e.target.files[0]); input.click(); }); if (!file) { throw new Error('未选择文件'); } console.log('3. 文件信息:', { name: file.name, type: file.type, size: file.size }); // 3. 构建上传URL const uploadUrl = `https://mooc-upload-ans.chaoxing.com/ueditorupload/upload-image?uid=&enc2=${encMatch[1]}&t=${timeMatch[1]}&encode=utf-8`; console.log('4. 上传URL:', uploadUrl); // 4. 使用 FormData const formData = new FormData(); formData.append('id', 'WU_FILE_0'); formData.append('name', file.name); formData.append('type', file.type); formData.append('lastModifiedDate', new Date().toString()); formData.append('size', file.size); formData.append('upfile', file); // 5. 使用 fetch 发送请求 console.log('5. 开始上传...'); const uploadResponse = await fetch(uploadUrl, { method: 'POST', body: formData, headers: { 'Accept': '*/*', 'Accept-Language': 'zh-CN,zh;q=0.9' }, credentials: 'include' // 保证传递用户的cookie等凭证 }); if (!uploadResponse.ok) { throw new Error(`上传失败: HTTP ${uploadResponse.status} ${uploadResponse.statusText}`); } const result = await uploadResponse.json(); console.log('6. 上传结果:', result); if (result.state === 'SUCCESS' && result.url) { console.log('7. 图片链接:', result.url); // 显示上传成功的图片链接 const img = document.createElement('img'); img.src = result.url; img.style.cssText = ` position: fixed; top: 10px; right: 10px; max-width: 300px; z-index: 9999; border: 2px solid #333; border-radius: 4px; box-shadow: 0 0 10px rgba(0,0,0,0.3); `; document.body.appendChild(img); // 创建复制链接按钮 const copyBtn = document.createElement('button'); copyBtn.textContent = '复制链接'; copyBtn.style.cssText = ` position: fixed; top: 20px; right: 320px; padding: 5px 10px; background: #4CAF50; color: white; border: none; border-radius: 4px; cursor: pointer; z-index: 9999; `; copyBtn.onclick = () => { navigator.clipboard.writeText(result.url) .then(() => { copyBtn.textContent = '已复制!'; setTimeout(() => copyBtn.textContent = '复制链接', 2000); }) .catch(err => console.error('复制失败:', err)); }; document.body.appendChild(copyBtn); } else { throw new Error(result.msg || '上传失败，未返回图片链接'); } } catch (error) { console.error('错误:', { message: error.message, stack: error.stack }); } })();
